const SHEET_NAME = "Ark1"; function getSheet() { return SpreadsheetApp .getActiveSpreadsheet() .getSheetByName(SHEET_NAME); } /** * Hent alle svar */ function doGet() { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); const rows = data.slice(1); const result = rows.map(r => ({ name: r[0], w41: r[1], w42: r[2], w43: r[3], w44: r[4] })); return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } /** * Tilføj nyt svar */ function doPost(e) { const sheet = getSheet(); const body = JSON.parse(e.postData.contents); sheet.appendRow([ body.name, body.w41, body.w42, body.w43, body.w44 ]); return ContentService.createTextOutput("OK"); }
